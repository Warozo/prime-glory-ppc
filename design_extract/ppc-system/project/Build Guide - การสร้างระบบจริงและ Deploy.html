หน้าเว็บ (build เป็นไฟล์ static) วางบน Cloudflare Pages/Netlify — เรียกข้อมูลผ่าน Supabase client library โดยตรง ไม่ต้องมี server กลางที่ต้องดูแลเอง business logic ที่ซับซ้อน (จองของ/เบิก FIFO/ปิดล็อต) เขียนเป็น SQL Function ใน Supabase แล้วเรียกผ่าน rpc()

3 โครงสร้างฐานข้อมูล

3.1 ข้อมูลหลัก (Master Data)

-- สินค้า: วัตถุดิบ (rm) และสินค้าสำเร็จรูป (fg) รวมในตารางเดียว
create table items (
  code        text primary key,
  type        text not null check (type in ('rm','fg')),
  name_th     text not null,
  name_en     text,
  unit        text not null,
  category    text,
  status      text default 'active',
  created_at  timestamptz default now()
);

create table customers ( id bigint generated always as identity primary key, name text not null );
create table suppliers ( id bigint generated always as identity primary key, name text not null );

create table production_lines (
  id text primary key, name text not null, manpower int, daily_cap int
);

-- BOM (สูตรการผลิต) — 1 fg : หลายบรรทัดวัตถุดิบ
create table boms (
  id bigint generated always as identity primary key,
  fg_code text references items(code), version text, active boolean default true
);
create table bom_lines (
  id bigint generated always as identity primary key,
  bom_id bigint references boms(id) on delete cascade,
  rm_code text references items(code), qty numeric not null, unit text
);

-- เทมเพลตกระบวนการผลิต (workflow) + ขั้นตอน
create table workflows (
  id text primary key, name text, line_id text references production_lines(id)
);
create table workflow_steps (
  id bigint generated always as identity primary key,
  workflow_id text references workflows(id) on delete cascade,
  step_key text, name_th text, name_en text, seq int, duration_h numeric
);

3.2 ออเดอร์ & การผลิต

-- คำสั่งซื้อลูกค้า → ความต้องการผลิต (สถานะ 5 ขั้น)
create table orders (
  id text primary key,                -- เช่น SO-2452
  customer_id bigint references customers(id),
  fg_code text references items(code),
  qty int not null,
  due_date date,
  priority text default 'med',
  status text default 'request'
    check (status in ('request','waiting','reserved','scheduled','completed')),
  created_at timestamptz default now()
);

-- ใบสั่งผลิต (เปิดเมื่อจองวัตถุดิบสำเร็จ)
create table production_orders (
  id text primary key,                -- เช่น PO-7781
  order_id text references orders(id),
  fg_code text references items(code),
  qty int, line_id text references production_lines(id),
  workflow_id text references workflows(id),
  start_date date, days int,
  status text default 'reserved'
    check (status in ('reserved','scheduled','inprogress','completed'))
);

-- การจองวัตถุดิบต่อออเดอร์
create table reservations (
  id bigint generated always as identity primary key,
  order_id text references orders(id),
  rm_code text references items(code),
  qty_required numeric, qty_reserved numeric default 0
);

3.3 คลังวัตถุดิบ & WIP & สินค้าสำเร็จรูป

-- ล็อตวัตถุดิบ (FIFO ตามวันหมดอายุ) — remaining = คงเหลือจริง
create table lots (
  id bigint generated always as identity primary key,
  rm_code text references items(code),
  supplier_id bigint references suppliers(id),
  qty numeric, remaining numeric, lot_no text,
  expiry date, received_at date default now()
);
create table material_receipts (
  id bigint generated always as identity primary key,
  supplier_id bigint, rm_code text, qty numeric, lot_no text, expiry date,
  received_at timestamptz default now()
);
create table material_issues (
  id bigint generated always as identity primary key,
  production_order_id text, rm_code text, lot_id bigint, qty numeric,
  reason text, issued_at timestamptz default now()
);

-- งานระหว่างผลิตต่อสถานี (cum_out = ผลผลิตสะสมที่สถานีนั้น)
create table wip_stations (
  id bigint generated always as identity primary key,
  production_order_id text references production_orders(id),
  step_key text, seq int, cum_out numeric default 0
);
create table output_logs (
  id bigint generated always as identity primary key,
  production_order_id text, step_key text, qty numeric,
  logged_at timestamptz default now()
);

-- สินค้าสำเร็จรูป: รอ QC → เข้าคลัง
create table fg_receipts (
  id bigint generated always as identity primary key,
  production_order_id text, fg_code text, qty int, lot_no text,
  status text default 'pending'
    check (status in ('pending','accepted','rejected')),
  completed_at timestamptz default now()
);
create table fg_stock (
  id bigint generated always as identity primary key,
  fg_code text references items(code), lot_no text, qty int, expiry date
);

4 ระบบล็อกอิน & สิทธิ์ (RLS)

Supabase Auth จัดการล็อกอิน — เราเพิ่มตาราง profiles เก็บ role แล้วใช้ Row Level Security คุมสิทธิ์

-- โปรไฟล์ผู้ใช้ ผูกกับระบบ auth ของ Supabase
create table profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  full_name text, email text,
  role text default 'production'
    check (role in ('admin','ppc','warehouse','production','management')),
  status text default 'active'
);

-- helper: อ่าน role ของผู้ใช้ปัจจุบัน
create function my_role() returns text language sql stable as $$
  select role from profiles where id = auth.uid()
$$;

-- เปิด RLS แล้วตั้ง policy (ตัวอย่าง: ตาราง orders)
alter table orders enable row level security;

-- ทุกคนที่ล็อกอินอ่านได้
create policy "orders_read" on orders for select
  using (auth.uid() is not null);

-- เฉพาะ ppc/admin แก้ไขออเดอร์ได้
create policy "orders_write" on orders for all
  using (my_role() in ('ppc','admin'))
  with check (my_role() in ('ppc','admin'));

ตารางสิทธิ์ตาม role (ใช้เป็นแนวเขียน policy แต่ละตาราง)

5 Business Logic (SQL Functions)

logic สำคัญต้องอยู่ฝั่ง server เพื่อกันข้อมูลพัง — เขียนเป็น function แล้วเรียกจากหน้าเว็บด้วย supabase.rpc('ชื่อ', {...})

-- จองวัตถุดิบ: เช็ค BOM กับสต็อก ถ้าครบ → สร้าง reservations + เลื่อนสถานะ
create function reserve_materials(p_order text)
returns text language plpgsql as $$
declare shortage int;
begin
  -- มีวัตถุดิบขาดไหม (need = qty * bom_line.qty เทียบ remaining รวม)
  select count(*) into shortage from (
    select bl.rm_code,
      sum(bl.qty) * o.qty as need,
      coalesce((select sum(remaining) from lots where rm_code=bl.rm_code),0) as oh
    from orders o
    join boms b on b.fg_code=o.fg_code and b.active
    join bom_lines bl on bl.bom_id=b.id
    where o.id=p_order group by bl.rm_code, o.qty
  ) x where need > oh;

  if shortage > 0 then return 'SHORTAGE'; end if;

  update orders set status='reserved' where id=p_order;
  -- (สร้าง production_order + reservations ที่นี่)
  return 'OK';
end $$;

-- เบิกวัตถุดิบตาม BOM แบบ FIFO (หักล็อตที่หมดอายุก่อน)
create function issue_materials(p_po text)
returns void language plpgsql as $$
declare r record; remain numeric; l record;
begin
  for r in
    select bl.rm_code, sum(bl.qty)*po.qty as need
    from production_orders po
    join boms b on b.fg_code=po.fg_code and b.active
    join bom_lines bl on bl.bom_id=b.id
    where po.id=p_po group by bl.rm_code, po.qty
  loop
    remain := r.need;
    for l in select * from lots
        where rm_code=r.rm_code and remaining>0 order by expiry loop
      exit when remain <= 0;
      update lots set remaining = remaining - least(remaining, remain)
        where id=l.id;
      remain := remain - least(l.remaining, remain);
    end loop;
  end loop;
  update production_orders set status='inprogress' where id=p_po;
end $$;

ฟังก์ชันที่เหลือทำแบบเดียวกัน: report_output(po, step, qty) อัปเดต WIP + ปิดล็อตอัตโนมัติ · accept_fg(receipt_id) ย้ายเข้า fg_stock · ป้อน prompt ให้ AI เขียนตามรูปแบบข้างบนได้

6 แผนสร้างด้วย AI (ทีละเฟส)

สั่ง AI ทีละขั้นตามลำดับนี้ — แต่ละเฟสทดสอบให้ผ่านก่อนไปต่อ อย่ารวบสั่งทีเดียว

1

ตั้งโปรเจกต์ + เชื่อม Supabase

สร้างโปรเจกต์ React และต่อกับ Supabase พร้อมหน้าล็อกอิน

สร้างโปรเจกต์ Vite + React + TypeScript ติดตั้ง @supabase/supabase-js ตั้งค่า client จาก env var ทำหน้า Login ด้วย Supabase Auth (email/password) และ route guard ให้เข้าได้เมื่อล็อกอินแล้ว
2

สร้างฐานข้อมูล + Seed

รัน SQL ในหมวด 3–5 บน Supabase แล้วใส่ข้อมูลตัวอย่าง

ใน Supabase SQL Editor รันสคริปต์สร้างตารางทั้งหมด จากนั้นเขียน seed.sql ใส่ข้อมูลตัวอย่าง: items, BOM ของ Rose Serum 5g, ลูกค้า, สายการผลิต A/B/C และ workflow
3

ย้าย UI เดิมเข้าโปรเจกต์

นำดีไซน์ navy/blue + i18n ไทย/อังกฤษ + โครงเมนูจากต้นแบบมาใช้

ใช้ไฟล์ styles.css และ dictionary ไทย/อังกฤษจากต้นแบบ Prime Glory สร้าง layout: sidebar + topbar + ตัวสลับภาษา TH/EN (default ไทย) + ตัวสลับ role
4

ต่อข้อมูลจริงทีละโมดูล

เริ่มจากอ่านข้อมูล (ตาราง/การ์ด) ก่อนเขียน

ทำหน้า Item Master และ Customer Orders ให้อ่าน/เพิ่ม/แก้ผ่าน Supabase จริง ใช้ React Query (@tanstack/react-query) จัดการ data fetching และ cache
5

ต่อ business logic (RPC)

เชื่อมปุ่มจอง/เบิก/รายงานผล/รับเข้า เข้ากับ SQL functions

เชื่อมปุ่ม "จองวัตถุดิบ" เรียก supabase.rpc('reserve_materials') ถ้าได้ SHORTAGE ให้โชว์วัตถุดิบที่ขาด ทำแบบเดียวกันกับเบิก/รายงานผลผลิต/รับเข้า FG
6

Realtime + ขัดเกลา

ให้ตัวเลข/แดชบอร์ดอัปเดตสดเมื่อมีคนอื่นแก้

เปิด Supabase Realtime ให้แดชบอร์ดและ WIP board อัปเดตอัตโนมัติเมื่อข้อมูลเปลี่ยน เพิ่ม loading/empty/error states และ toast แจ้งผล

7 ขั้นตอน Deploy

A ขึ้น Cloudflare Pages

  • push โค้ดขึ้น GitHub
  • Cloudflare → Pages → เชื่อม repo
  • Build command: npm run build
  • Output: dist
  • ใส่ env: VITE_SUPABASE_URL, VITE_SUPABASE_ANON_KEY
  • กด Deploy → ได้ลิงก์ *.pages.dev

B หรือ Netlify

  • เชื่อม GitHub repo เดียวกัน
  • Build: npm run build · Publish: dist
  • ใส่ env เหมือนกัน
  • ได้ลิงก์ *.netlify.app
  • ผูกโดเมนตัวเองได้ภายหลัง (ฟรี)
Supabase ไม่ต้อง deploy — เป็นบริการ cloud อยู่แล้ว แค่สร้างโปรเจกต์ฟรี แล้วเอา URL + anon key มาใส่ใน env ของหน้าเว็บ ทุกครั้งที่ push GitHub ระบบจะ build + deploy ใหม่อัตโนมัติ
